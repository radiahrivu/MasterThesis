CREATE TABLE "ExperimentResult" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"userId"	INTEGER,
	"settingId"	INTEGER,
	"method"	INTEGER,
	"arousalScale"	INTEGER,
	"valenceScale"	INTEGER,
	"dominanceScale"	INTEGER,
	"timestamp"	TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	"counter"	INTEGER,
	"clip"	INTEGER
);

CREATE TABLE "ExperimentSetting" (
	"id"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"userId"	INTEGER,
	"environment"	INTEGER,
	"emotion"	INTEGER,
	"sequence"	INTEGER,
	"videoClip"	INTEGER,
	"audioClip"	INTEGER,
	"imageClip"	INTEGER
);


CREATE TABLE "User" (
	"id"	INTEGER,
	"gender"	INTEGER,
	"age"	INTEGER,
	"profession"	TEXT,
	"vrExperience"	INTEGER,
	PRIMARY KEY("id")
);
